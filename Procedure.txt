CREATE OR REPLACE 
PROCEDURE EDITADDRESS(GADDRESS IN VARCHAR2,GID IN NUMBER) IS

BEGIN

END;

CREATE OR REPLACE 
PROCEDURE P1(FITEM IN NUMBER,pid OUT NUMBER) IS

BEGIN

SELECT PRICE INTO pid FROM FOOD WHERE FOOD_ID=FITEM;

END;

//

CREATE OR REPLACE 
PROCEDURE P2(INID IN NUMBER,AMO IN NUMBER) IS

BEGIN

UPDATE BILL SET AMOUNT=AMO WHERE INVOICE_ID=INID;

END;

//

CREATE OR REPLACE 
PROCEDURE P3(GNAME IN VARCHAR2,GID IN NUMBER,GMAIL IN VARCHAR2,GPHONE IN VARCHAR2,
GADDRESS IN VARCHAR2,GCITY IN VARCHAR2,GCOUNTRY IN VARCHAR2,GPASSWORD IN VARCHAR2) IS

BEGIN

UPDATE GUEST SET NAME=GNAME,EMAIL=GMAIL,PHONE=GPHONE,ADDRESS=GADDRESS,
CITY=GCITY,COUNTRY=GCOUNTRY,PASSWORD=GPASSWORD WHERE GUEST_ID=GID;

END;
